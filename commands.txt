Install:
    npm init -y
    npm install --save-dev typescript ts-node @types/node
    npm install @babel/core @babel/cli @babel/preset-typescript @babel/parser @babel/traverse @babel/types @babel/generator
    npm i --save-dev @types/babel__core

Run:
    npx tsc
    npx babel ./TestMaterial/Input/input.ts --out-file ./TestMaterial/Output/output.js
